import numpy as np

# Generate sample data

# Prepare features and target
X = np.array([
    # Your feature data here
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
])

y = np.array([10, 20, 30])

# Train model
result = ml_manager.train_model(
    algorithm_name='random_forest',
    task_type='regression',
    X=X,
    y=y
)

# Make predictions

    predictions = ml_manager.predict(
        model_key=result['model_key'],
        X=np.array([[1, 2, 3]])
    )
    print(f"Predictions: {predictions['predictions']}")

# Save model
ml_manager.save_model(
    model_key=result['model_key'],
    filepath='model.pkl'
)
